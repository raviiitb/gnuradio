00069   void step ()
00070   {
00071     phase += phase_inc;
00072     if (fabs (phase) > M_PI){
00073 
00074       while (phase > M_PI)
00075         phase -= 2*M_PI;
00076 
00077       while (phase < -M_PI)
00078         phase += 2*M_PI;
00079     }
00080   }

00082   void step (int n)
00083   {
00084     phase += phase_inc * n;

00086 
00087       while (phase > M_PI)
00088         phase -= 2*M_PI;
00089 
00090       while (phase < -M_PI)
00091         phase += 2*M_PI;
00092     }
00093   }
