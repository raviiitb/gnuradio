00118 inline gruel::high_res_timer_type gruel::high_res_timer_epoch(void){
00119     static const double hrt_ticks_per_utc_ticks = gruel::high_res_timer_tps()/double(boost::posix_time::time_duration::ticks_per_second());
00120     boost::posix_time::time_duration utc = boost::posix_time::microsec_clock::universal_time() - boost::posix_time::from_time_t(0);
00121     return gruel::high_res_timer_now() - utc.ticks()*hrt_ticks_per_utc_ticks;
