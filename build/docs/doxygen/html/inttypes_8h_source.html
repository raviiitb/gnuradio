00277 {
00278    imaxdiv_t result;
00279 
00280    result.quot = numer / denom;
00281    result.rem = numer % denom;
00282 
00283    if (numer < 0 && result.rem > 0) {
00284       // did division wrong; must fix up
00285       ++result.quot;
